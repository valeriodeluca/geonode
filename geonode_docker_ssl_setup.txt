  django:
    build: .
    # Loading the app is defined here to allow for

    volumes:
      - '.:/usr/src/app'
    environment:
      - DEBUG=False
      - GEONODE_LB_HOST_IP=Server_IP
      - GEONODE_LB_PORT=80
      - SITEURL=http://Server_IP/
      - ALLOWED_HOSTS=['*']
      - GEOSERVER_PUBLIC_LOCATION=http://Server_IP/geoserver/
      - GEOSERVER_WEB_UI_LOCATION=http://Server_IP/geoserver/

  geoserver:
    environment:
      - GEONODE_LB_HOST_IP=Server_IP
      - GEONODE_LB_PORT=80
